#include <iostream>
#include <vector>

using namespace std;

int main() {
  // Define a 2x3 matrix
  vector<vector<int>> matrix = {{1, 2, 3}, {4, 5, 6}};

  // Print the matrix
  for (int i = 0; i < matrix.size(); i++) {
    for (int j = 0; j < matrix[i].size(); j++) {
      cout << matrix[i][j] << " ";
    }
    cout << endl;
  }

  return 0;
}
